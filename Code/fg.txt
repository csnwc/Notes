<MAIN>         ::= "{" <INSTRCTLST>
<INSTRCTLST>   ::= <INSTRUCTION><INSTRCTLST> | "}"
<INSTRUCTION>  ::= <FD> | <LT> | <RT> | <DO> | <SET>
<FD>           ::= "FD" <VARNUM>
<LT>           ::= "LT" <VARNUM>
<RT>           ::= "RT" <VARNUM>
<DO>           ::= "DO" <VAR> "FROM" <VARNUM> "TO" <VARNUM> "{" <INSTRCLST>
<VAR>          ::= [A-Z]
<VARNUM>       ::= number | <VAR>
<SET>          ::= "SET" <VAR> ":=" <POLISH>
<POLISH>       ::= <OP><POLISH> | <VARNUM><POLISH> | ";"
<OP>           ::= "+ | "-" | "*" | "/"
